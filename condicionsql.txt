SELECT horasdestajo.codtraba, straba.nomtraba, horasdestajo.fechahora, horasdestajo.codvarie, variedades.nomvarie, horasdestajo.codforfait, forfaits.nomconfe, horasdestajo.numcajon, horasdestajo.kilos, horasdestajo.importe,  horasdestajo.horas  FROM horasdestajo, straba, variedades, forfaits  WHERE horasdestajo.codtraba = straba.codtraba and  horasdestajo.codvarie = variedades.codvarie and  horasdestajo.codforfait = forfaits.codforfait  AND (horasdestajo.codtraba  =  000080) ORDER BY  horasdestajo.fechahora desc, horasdestajo.codtraba
