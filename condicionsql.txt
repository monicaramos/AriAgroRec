select rhisfruta.numalbar from rhisfruta LEFT JOIN rhisfruta_entradas ON rhisfruta.numalbar=rhisfruta_entradas.numalbar  WHERE (rhisfruta.fecalbar  =  '2018-09-29') and rhisfruta.codvarie in (select codvarie from variedades, productos, grupopro where grupopro.codgrupo <> 5 and grupopro.codgrupo <> 6  and variedades.codprodu = productos.codprodu and productos.codgrupo = grupopro.codgrupo )  GROUP BY rhisfruta.numalbar  ORDER BY rhisfruta.numalbar
