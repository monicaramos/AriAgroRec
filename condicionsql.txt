SELECT horas.codtraba, straba.nomtraba, horas.fechahora, horas.codalmac, salmpr.nomalmac, horas.horasdia, horas.horasproduc, horas.compleme, horas.horasext,  horas.codvarie, variedades.nomvarie, horas.codforfait, forfaits.nomconfe, horas.fecharec,  horas.pasaridoc,  IF(pasaridoc=1,'*','') as pasari, horas.intconta,  IF(intconta=1,'*','') as intcon  FROM straba, salmpr, variedades, horas LEFT JOIN  forfaits ON horas.codforfait = forfaits.codforfait  WHERE horas.codtraba = straba.codtraba and  horas.codalmac = salmpr.codalmac and  horas.codvarie = variedades.codvarie  and (1=1)   ORDER BY  horas.fechahora desc, horas.codtraba, horas.codalmac, horas.codvarie
